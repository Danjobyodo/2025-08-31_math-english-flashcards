        // --- アプリケーションロジック ---

        const cardContainer = document.getElementById('card-container');

        const cardInner = document.getElementById('card-inner');

        const englishWordEl = document.getElementById('english-word');

        const japaneseWordEl = document.getElementById('japanese-word');

        let currentWord = null;

        let isFlipped = false;

        let usedIndices = new Set();


        // ランダムに単語を取得

        function getRandomWord() {

            if (usedIndices.size === words.length) {

                usedIndices.clear(); // 全ての単語を表示したらリセット

            }

            let randomIndex;

            do {

                randomIndex = Math.floor(Math.random() * words.length);

            } while (usedIndices.has(randomIndex));

            

            usedIndices.add(randomIndex);

            return words[randomIndex];

        }


        // カードの表示を更新

        function updateCard() {

            currentWord = getRandomWord();

            englishWordEl.textContent = currentWord.english;

            japaneseWordEl.textContent = currentWord.japanese;

            

            // カードを表に戻す

            if (isFlipped) {

                cardInner.classList.remove('rotate-y-180');

                isFlipped = false;

            }

        }


        // カードを裏返す

        function flipCard() {

            isFlipped = !isFlipped;

        }
